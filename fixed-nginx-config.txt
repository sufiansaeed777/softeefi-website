server {
    server_name softeefi.co.uk www.softeefi.co.uk;

    # Gzip Compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml application/javascript application/json application/xml+rss application/rss+xml application/atom+xml image/svg+xml application/x-javascript application/x-font-ttf application/vnd.ms-fontobject font/opentype;
    gzip_comp_level 6;

    # === REDIRECTS FOR OLD URLs (Fixes Google 404s) ===
    location = /free-reports {
        return 301 https://softeefi.co.uk/free-learning;
    }

    location = /services {
        return 301 https://softeefi.co.uk/;
    }

    location = /ai-solutions {
        return 301 https://softeefi.co.uk/ai-integration;
    }

    location = /web-development {
        return 301 https://softeefi.co.uk/websites-and-apps;
    }

    location = /digital-marketing-services {
        return 301 https://softeefi.co.uk/digital-marketing;
    }

    location = /portfolio {
        return 301 https://softeefi.co.uk/projects;
    }

    location = /testimonials {
        return 301 https://softeefi.co.uk/;
    }

    location = /resources {
        return 301 https://softeefi.co.uk/blog;
    }

    location = /careers {
        return 301 https://softeefi.co.uk/contact;
    }

    location = /partners {
        return 301 https://softeefi.co.uk/contact;
    }

    location = /case-studies {
        return 301 https://softeefi.co.uk/projects;
    }

    location = /get-started {
        return 301 https://softeefi.co.uk/contact;
    }

    location = /solutions {
        return 301 https://softeefi.co.uk/;
    }

    location = /services/digital-art {
        return 301 https://softeefi.co.uk/services/digital-art-nfts;
    }

    location = /services/cloud-services {
        return 301 https://softeefi.co.uk/services/cloud-infrastructure;
    }

    location = /services/video-production {
        return 301 https://softeefi.co.uk/services/graphic-design;
    }

    location = /projects/digital-marketing-projects {
        return 301 https://softeefi.co.uk/projects/digital-marketing;
    }

    location = /projects/cloud-projects {
        return 301 https://softeefi.co.uk/projects/cloud-infrastructure;
    }
    
    location = /free-reports/websites-and-apps {
        return 301 https://softeefi.co.uk/free-learning;
    }

    location = /blog/6 {
        return 301 https://softeefi.co.uk/blog;
    }

    location = /blog/4 {
        return 301 https://softeefi.co.uk/blog;
    }
    # === END REDIRECTS ===

    # Frontend with caching
    location / {
        root /var/www/softeefi/frontend/build;
        try_files $uri $uri/ /index.html;

        # Security headers
        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header X-XSS-Protection "1; mode=block" always;
    }

    # Static assets caching
    location ~* \.(jpg|jpeg|png|gif|webp|svg|ico|css|js|woff|woff2|ttf|otf|eot|mp4|mov)$ {
        root /var/www/softeefi/frontend/build;
        expires 30d;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # Backend API
    location /api {
        proxy_pass http://localhost:4000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    client_max_body_size 10M;

    # SSL Configuration
    listen 443 ssl http2;
    ssl_certificate /etc/letsencrypt/live/softeefi.co.uk/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/softeefi.co.uk/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    listen 80;
    server_name softeefi.co.uk www.softeefi.co.uk;
    return 301 https://$host$request_uri;
}
